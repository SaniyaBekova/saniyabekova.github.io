My original research question was: “Identifying bots in social media content.”

However, I am planning to revise this question due to several challenges I have encountered.

First, I attempted a semi-supervised learning approach using labeled data from an external source.

I have contacted the data owners via email to request access and am currently waiting for their response.

If I receive the data, I plan to proceed with that method.

Second, I experimented with unsupervised learning, specifically using DBSCAN clustering and Isolation Forest to detect anomalies.

DBSCAN did not yield meaningful results — it produced only a single cluster.

In contrast, Isolation Forest classified 273 comments as human-generated and 31 as potential bot-generated.

import json
import re
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.cluster import DBSCAN
from sklearn.ensemble import IsolationForest
from sklearn.decomposition import PCA

def preprocess_text(text):
    text = text.lower()
    text = re.sub(r"@\w+", "", text)
    text = re.sub(r"[^a-zA-Zа-яА-ЯёЁ\s]", "", text)
    text = re.sub(r"\s+", " ", text).strip()
    return text

with open("data/tweets_all.json", 'r', encoding='utf-8') as f:
    data = json.load(f)

df = pd.DataFrame(data)
df['clean_text'] = df['text'].apply(preprocess_text)

vectorizer = TfidfVectorizer(max_features=1000)
X = vectorizer.fit_transform(df['clean_text'])

dbscan = DBSCAN(eps=0.5, min_samples=5, metric='cosine')
df['cluster'] = dbscan.fit_predict(X)

iso_forest = IsolationForest(contamination=0.1, random_state=42)
df['anomaly'] = iso_forest.fit_predict(X.toarray())

pca = PCA(n_components=2)
X_pca = pca.fit_transform(X.toarray())
plt.figure(figsize=(8, 6))
plt.scatter(X_pca[:, 0], X_pca[:, 1], c=df['cluster'], cmap='viridis', s=5)
plt.title("DBSCAN Clustering of Comments")
plt.xlabel("PC1")
plt.ylabel("PC2")
plt.colorbar(label="Cluster")
plt.show()

import json
import re
import pandas as pd
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset
from sentence_transformers import SentenceTransformer
from sklearn.model_selection import train_test_split
import matplotlib.pyplot as plt

# === Text Preprocessing ===
def preprocess_text(text):
    text = text.lower()
    text = re.sub(r"http\S+", "", text)
    text = re.sub(r"@\w+", "", text)
    text = re.sub(r"[^a-zA-Zа-яА-ЯёЁ\s]", "", text)
    return re.sub(r"\s+", " ", text).strip()

# === Autoencoder Definition ===
class Autoencoder(nn.Module):
    def __init__(self, input_dim):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, 256),
            nn.ReLU(),
            nn.Linear(256, 64),
            nn.ReLU()
        )
        self.decoder = nn.Sequential(
            nn.Linear(64, 256),
            nn.ReLU(),
            nn.Linear(256, input_dim),
        )

    def forward(self, x):
        return self.decoder(self.encoder(x))

# === Load and Preprocess Data ===
with open("data/tweets_all.json", 'r', encoding='utf-8') as f:
    raw_data = json.load(f)

df = pd.DataFrame(raw_data)
df['clean_text'] = df['text'].apply(preprocess_text)

# === Generate BERT Embeddings ===
print("🔄 Generating Sentence-BERT embeddings...")
model_name = 'paraphrase-MiniLM-L6-v2'
embedder = SentenceTransformer(model_name)
embeddings = embedder.encode(df['clean_text'].tolist(), batch_size=32, show_progress_bar=True)

# === Train Autoencoder ===
X = torch.tensor(embeddings, dtype=torch.float32)
X_train, X_test = train_test_split(X, test_size=0.1, random_state=42)
train_loader = DataLoader(TensorDataset(X_train), batch_size=64, shuffle=True)

autoencoder = Autoencoder(input_dim=X.shape[1])
criterion = nn.MSELoss()
optimizer = torch.optim.Adam(autoencoder.parameters(), lr=1e-3)

EPOCHS = 25
print("🚀 Training autoencoder...")
for epoch in range(EPOCHS):
    autoencoder.train()
    total_loss = 0
    for batch in train_loader:
        inputs = batch[0]
        outputs = autoencoder(inputs)
        loss = criterion(outputs, inputs)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        total_loss += loss.item()
    print(f"Epoch {epoch+1}/{EPOCHS}, Loss: {total_loss:.4f}")

# === Compute Reconstruction Error ===
autoencoder.eval()
with torch.no_grad():
    reconstructed = autoencoder(X)
    mse = torch.mean((X - reconstructed) ** 2, dim=1).numpy()

df['reconstruction_error'] = mse
threshold = pd.Series(mse).quantile(0.90)
df['is_bot'] = df['reconstruction_error'] > threshold

# === Save Results ===
df.to_csv("bert_bot_detection_results.csv", index=False)
print("✅ Results saved to 'bert_bot_detection_results.csv'")

# === Visualization ===
plt.hist(df['reconstruction_error'], bins=60)
plt.axvline(threshold, color='red', linestyle='--', label='Bot Threshold')
plt.title("Sentence-BERT Autoencoder — Reconstruction Error")
plt.xlabel("Reconstruction Error")
plt.ylabel("Frequency")
plt.legend()
plt.tight_layout()
plt.show()
